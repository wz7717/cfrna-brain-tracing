cfRNA-BrainTrace v0.1.6 Jupyter Lab 人工验证复现工作流

按论文当前 submission route 复现 Bo2023、AHBA、TCGA/BraTS 和 GSE189919 验证数字。每个模块都包含输入文件、运行代码、输出文件和目标数值断言。

0. 使用方式

先运行“环境启动页”的 PowerShell 代码启动 Jupyter Lab。然后新建一个 Python notebook，按本 HTML 中的单元顺序复制运行。所有结果写入 output/v016_jupyter_validation，不会修改论文主稿、Supplementary 或 source data。

双模式Public audit 模式默认可在 GitHub clone 后跑通，只核对公开代码、模型 artifact 和论文 source data。Controlled-data validation 模式会在检测到本地授权 Bo2023/AHBA/TCGA/GSE 输入文件时自动启用原始数据级重跑。

当前论文路线projected-VSD 只用于 Network Top3 beam；resolution-group 和 exact-region 使用 logCPM-compatible local reranking。不要把旧 baseline 或 MRI truth 旁路线当成 Table S4 当前 submission route。

def run_cmd(args: list[str], outdir: Path, log_name: str = "run.log", timeout: int | None = None) -> subprocess.CompletedProcess:
    outdir.mkdir(parents=True, exist_ok=True)
    log_path = outdir / log_name
    print("Running:", " ".join(map(str, args)))
    started = time.time()
    proc = subprocess.run(
        [str(x) for x in args],
        cwd=ROOT,
        text=True,
        stdout=subprocess.PIPE,
        stderr=subprocess.STDOUT,
        timeout=timeout,
    )
    log_path.write_text(proc.stdout, encoding="utf-8", errors="replace")
    print(proc.stdout[-4000:])
    print("Exit code:", proc.returncode, "elapsed_sec:", round(time.time() - started, 1))
    if proc.returncode != 0:
        raise RuntimeError(f"Command failed, see {log_path}")
    return proc

def pct(value: float) -> str:
    return f"{100 * float(value):.2f}%"

def assert_close(actual: float, expected: float, tol: float = 5e-5, label: str = "") -> None:
    if abs(float(actual) - float(expected)) > tol:
        raise AssertionError(f"{label}: actual={actual}, expected={expected}")
    print("OK", label, pct(actual))

def skip_controlled(route_name: str, missing: list[Path]) -> None:
    print(f"SKIP controlled-data validation: {route_name}")
    print("缺少本地授权/大体量输入文件。Public audit 将只核对论文 source_data 和代码路径。")
    for path in missing:
        print("missing:", path)

def missing_paths(paths: list[Path]) -> list[Path]:
    return [path for path in paths if not path.exists()]

import matplotlib.pyplot as plt
from IPython.display import display, Image

def save_display_table(df: pd.DataFrame, out_path: Path, title: str = "") -> Path:
    out_path.parent.mkdir(parents=True, exist_ok=True)
    df.to_csv(out_path, index=False, encoding="utf-8-sig")
    if title:
        print(title)
    display(df)
    print("表格已保存:", out_path)
    return out_path

def save_metric_barplot(
    rows: list[dict],
    out_path: Path,
    title: str,
    x_col: str = "metric",
    y_col: str = "actual",
    target_col: str = "target",
) -> Path:
    out_path.parent.mkdir(parents=True, exist_ok=True)
    frame = pd.DataFrame(rows)
    fig, ax = plt.subplots(figsize=(max(7, 1.35 * len(frame)), 4.4))
    bars = ax.bar(frame[x_col], frame[y_col], color="#2f6fb2", label="actual")
    if target_col in frame.columns:
        ax.scatter(frame[x_col], frame[target_col], color="#b42318", zorder=3, label="target")
    ax.set_ylim(0, max(1.0, float(frame[[y_col, target_col]].max().max()) * 1.15 if target_col in frame.columns else float(frame[y_col].max()) * 1.15))
    ax.set_ylabel("accuracy / fraction")
    ax.set_title(title)
    ax.tick_params(axis="x", labelrotation=25)
    ax.bar_label(bars, labels=[f"{v:.2%}" for v in frame[y_col]], padding=3, fontsize=9)
    ax.legend(loc="upper right")
    fig.tight_layout()
    fig.savefig(out_path, dpi=180)
    plt.show()
    print("图已保存:", out_path)
    return out_path

def metric_table(rows: list[dict]) -> pd.DataFrame:
    frame = pd.DataFrame(rows)
    if {"actual", "target"}.issubset(frame.columns):
        frame["actual_percent"] = frame["actual"].map(lambda x: f"{x:.2%}")
        frame["target_percent"] = frame["target"].map(lambda x: f"{x:.2%}")
        frame["delta_percent_point"] = (frame["actual"] - frame["target"]).map(lambda x: f"{100*x:+.3f}")
    return frame
